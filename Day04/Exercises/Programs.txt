Logical Operators:
Guess the output before writing program and then validate your answer.

1.       int x = 5;
         System.out.println(x > 3 && x < 10);
         System.out.println(x > 3 || x < 10);
         System.out.println(!(x > 3 && x < 10));

2.       int a = 25, b = -11, c = 0;
         System.out.println((a<c && b<c) && (a>c && b>c));
         System.out.println((a>c && b<c) && (a<c && b>c));
         
         System.out.println((a<c && b<c) || (a>c && b>c));
         System.out.println((a>c && b<c) || (a<c && b>c));
         
         System.out.println((a<c || b<c) && (a>c || b>c));
         System.out.println((a>c || b<c) && (a<c || b>c));
         
         System.out.println((a<c || b<c) || (a>c || b>c));
